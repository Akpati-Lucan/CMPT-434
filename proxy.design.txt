void *sender_thread() {

    while (1) {
        Dequeue the first message in the outgoing_messages queue and sends it to proxy server
    }
}

void *reciever_thread() {

    while (1) {
        recieve Messages from socket and enqueue it to the list of incoming_messages messages 
    }
}


void *main_server() {

    while (1) {
        grab MSG from incoming_messages
        switch (msg.label) {
        case NEW_SERVER:
            Construct A ADD_SERVER MSG for all known servers 
            enqueue them to outgoing messages

            Add new server details to the server table
        
        }

        default:

            Add MSG to outgoing_messages
        
    }

}
