void *heartbeat_thread() {

    while (1) {
        construct a MSG with hearbeat label and add to the list of outgoing_messages queue
        set the global heartbeat_ack flag to 0
        sleep for timeout time.

        if heartbeat_ack == 1
            if response is recieve sleep for some time
        else call election thread
            break - close heartbeat thread
    }
}


void *election_thread() {

    if number of votes >= N / 2

void *main_server() {

    while (1) {
        grab MSG from incoming_messages
        switch (msg.label) {

            case: NEW_LOG_VALUE { // For Leader Server 
                if the current process is the leader thread
                it constructs a MSG for all known servers with label appending and enqueue outgoing_messages
                start append_handler_thread(seq_number_of_log)
            }

            case: APPEND {
                Add msg.msg to the log 
                increment log_append_tracker[seq_number_of_log]
                construct a MSG with label APPEND_ACK and enqueue
                to outgoing messages
            }
            case: APPEND_ACK { // Only if Server is a leader
                Add 1 to the log_append_tracker for the seq_number of the message
            }
            case: REJECT { // For follower server
                clear log position for sequence number in MSG
                make sure seq_number_of_log is at correct position
            }
            case: COMMIT { // for follower server
                if the seq_number_of_log of personal log is less than seq number of commit MSG
                Create a reject message and add to outgoing queue
                Increment the seq_number_of_log
            }
            case: HEARTBEAT { // for leader server Only
                Construct A heartbeat_ack Message and enqueue to 
                outgoing_messages
            }
            case: HEARTBEAT_ACK { for follower server only
                set global heartbeat_ack to 1
            }
            case: VOTE {
                check if seq_number of sender of MSG is >= than ours if yes 
                vote for sender
                else don't vote
                Create a VOTE_FOR MSG and send to outgoing_messages
            }
            case: VOTE_FOR {
                if MSG.vote == 1 
                add 1 to number_of_votes
            }
            case: NEW_LEADER {
                Set the flag in the sever table of the new leader to 1
                and the flag of the old leader to zero
            }
            case: UPDATE_CHECK { // for leader server
                Send the current seq number of the keadders log
                Enqueue to outgong messages
            }
            case: UPDATE_SIGNAL { // For follower server
                Check your current seq_number_of_log and compare to the seq_number in the log
                if the seq numbers are the same fine
                else 
                send and update MSG containing your seq number
            }
            case: UPDATE { // For Leader Server 
                Send a UPDATE_ACK MSG to the specified follower server containing the log vaulue of the 
                seq number in the MSG
            }
            case: UPDATE_ACK { // For Follower Server
                Update your Log with the log value in the message from the leader 
                increment your seq_number_of_log
                Construct a UPDATE_CHECK MSG and enqueue to outgoing messages
            }
            case: ADD_SERVER { // For all servers
            // From Proxy Server
                ADD source_name and source_port to list of servers
                is-leader is by default zero
            }
        }
    }
}
